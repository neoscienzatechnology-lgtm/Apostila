1. Tipos, Variáveis e Coerção

JavaScript possui tipos primitivos (string, number, boolean, null, undefined, symbol, bigint) e tipos por referência (objects, arrays, functions). A linguagem faz coerção implícita em várias operações — importante para evitar bugs.

// Declarações
let nome = 'Maria';
const PI = 3.1415;
let idade = 28; // number
let ativo = true; // boolean
let dados = null; // null
let indefinido; // undefined

// Coerção de tipos
console.log('5' - 2); // 3 (string -> number)
console.log('5' + 2); // '52' (number -> string com +)

// typeof
console.log(typeof nome); // 'string'
console.log(typeof dados); // 'object' (historical JS quirk)

Observação: evite confiar em coerções implícitas em código crítico — prefira conversões explícitas (Number(), String(), Boolean()).

2. Operadores e Controle de Fluxo

// Operadores de comparação
console.log(1 == '1');  // true (coerção)
console.log(1 === '1'); // false (estritamente igual)

// if / else
if (idade >= 18) {
    console.log('Maior de idade');
} else {
    console.log('Menor de idade');
}

// switch
switch (new Date().getDay()) {
    case 0:
        console.log('Domingo');
        break;
    default:
        console.log('Outro dia');
}

// Loops
for (let i = 0; i < 5; i++) console.log(i);
let j = 0; while(j < 3) { console.log(j); j++; }

Use as estruturas que tornam o código mais legível; prefira iterações funcionais (map/filter/reduce) quando apropriado.

3. Funções — Declaração, Expressão e Arrow

Funções são cidadãos de primeira classe em JavaScript: podem ser armazenadas, passadas como argumento e retornadas.

// Declaração (hoisted)
function soma(a, b) {
    return a + b;
}

// Expressão de função
const sub = function(a, b) { return a - b; };

// Arrow function
const mult = (a, b) => a * b;

// Parâmetros padrão e rest
function log(msg = 'vazio', ...items) {
    console.log(msg, items);
}

// IIFE
(function() {
    console.log('executa imediatamente');
})();

Atenção ao this: arrow functions não possuem seu próprio this — isso altera comportamento em métodos e callbacks.

4. Escopo, Hoisting e Closures

Entender como o escopo funciona é essencial para evitar leaks e bugs difíceis de rastrear.

// Escopo de função / bloco
{
    let a = 1; // visível apenas dentro do bloco
}

// Hoisting
console.log(x); // undefined (devido ao hoisting de var)
var x = 10;

// Temporal Dead Zone
// console.log(y); // ReferenceError
let y = 20;

// Closure
function contador() {
    let count = 0;
    return function() {
        count++;
        return count;
    };
}
const inc = contador();
console.log(inc()); // 1
console.log(inc()); // 2

Closures são úteis para encapsular estado; prefira padrões que evitem exposição desnecessária do estado.

5. Boas práticas, erros e depuração

  • Use const por padrão; let apenas quando o valor muda.
  • Evite var — causa hoisting confuso.
  • Valide entradas e trate erros com try/catch quando necessário.
  • Console e breakpoints: prefira depuração com ferramentas do navegador para cenários complexos.
try {
    const data = JSON.parse('dados inválidos');
} catch (err) {
    console.error('Erro ao parsear JSON', err);
}

6. Exemplos Práticos

// Debounce simples
function debounce(fn, wait) {
    let timer;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => fn.apply(this, args), wait);
    };
}

// Uso de map/filter/reduce
const pessoas = [
    {nome: 'Ana', idade: 20},
    {nome: 'Bruno', idade: 30},
];
const nomes = pessoas.map(p => p.nome);
const somaIdades = pessoas.reduce((s, p) => s + p.idade, 0);
